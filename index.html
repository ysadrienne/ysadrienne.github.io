Lab 1B: Bluetooth

Task 1: ECHO command

The ECHO command sends a string to the Artemis board, in which the commputer in turn prints a modified string. tx_estring_value adds Robot says -> to the front of the string and appends a :) at the end of the string, and tx_characteristic_string sends it to the computer.

The following is the Arduino code: 


case ECHO:

    char char_arr[MAX_MSG_SIZE];
                        
    // Extract the next value from the command string as a character array
    success = robot_cmd.get_next_value(char_arr);
    if (!success)
        return;
    tx_estring_value.clear();
    tx_estring_value.append("Robot says -> ");
    tx_estring_value.append(char_arr);
    tx_estring_value.append(" :)");
    tx_characteristic_string.writeValue(tx_estring_value.c_str());
    Serial.println(tx_estring_value.c_str());
                                     
    break;

Task 2: SEND_THREE_FLOATS command

The SEND_THREE_FLOATS command sends three floats to the Artemis board, in which the commputer in turn extracts and prints the floats. Like the ECHO command, tx_estring_value modifies the string by appending the floats, and tx_characteristic_string sends it to the computer. The following is the Arduino code: 


case SEND_THREE_FLOATS:
    float float_a, float_b, float_c;
                        
    // Extract the next value from the command string as a float
    success = robot_cmd.get_next_value(float_a);
    if (!success)
        return;
                        
    // Extract the next value from the command string a float
    success = robot_cmd.get_next_value(float_b);
    if (!success)
        return;
                        
    // Extract the next value from the command string a float
    success = robot_cmd.get_next_value(float_c);
    if (!success)
        return;
                        
    tx_estring_value.clear();
    tx_estring_value.append("Three Floats: ");
    tx_estring_value.append((float)float_a);
    tx_estring_value.append(", ");
    tx_estring_value.append((float)float_b);
    tx_estring_value.append(", ");
    tx_estring_value.append((float)float_c);
    tx_characteristic_string.writeValue(tx_estring_value.c_str());
                        
    Serial.println(tx_estring_value.c_str());
                                     
    break;
